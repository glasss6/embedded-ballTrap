355 void CallbackAccelerometer(float x, float y, float z) {
356  if(!gTrapArmed) return; // If not armed (in grace period), do not check
357 #ifdef LOG_ACCEL_EVENTS
358  UART_printf(SUBSYSTEM_UART, "%f, %f, %f\r\n", x, y, z);
359 #endif
360  float normAccel = (fabs(x) + fabs(y) + fabs(z)) - 1; // Manhattan Distance
361  // Subtracting 1 to normalize the effect of gravity on acceleration
362  if (normAccel > COLLISION_THRESHOLD) { // Arbitrary threshold value (refer to documentation)
363 #ifdef LOG_ACCEL_EVENTS
364  UART_printf(SUBSYSTEM_UART, "Tripped!\r\n", x, y, z);
365 #endif
366  Thief_Trip(1); // Collision detected, so trip to thief module
367  gTrapArmed = 0;
368  //Accelerometer_Deinit();

405 void initClockTo16MHz()
406 {
407  increaseVCoreToLevel2();
408  UCSCTL3 |= SELREF_2; // Set DCO FLL reference = REFO
409  UCSCTL4 |= SELA_2; // Set ACLK = REFO
410  __bis_SR_register(SCG0); // Disable the FLL control loop
411  UCSCTL0 = 0x0000; // Set lowest possible DCOx, MODx
412  UCSCTL1 = DCORSEL_5; // Select DCO range 16MHz operation
413  UCSCTL2 = FLLD_0 + 487; // Set DCO Multiplier for 16MHz
414  // (N + 1) * FLLRef = Fdco
415  // (487 + 1) * 32768 = 16MHz
416  // Set FLL Div = fDCOCLK
417  __bic_SR_register(SCG0); // Enable the FLL control loop
418 
419  // Worst-case settling time for the DCO when the DCO range bits have been
420  // changed is n x 32 x 32 x f_MCLK / f_FLL_reference. See UCS chapter in 5xx
421  // UG for optimization.
422  // 32 x 32 x 16 MHz / 32,768 Hz = 500000 = MCLK cycles for DCO to settle
423  __delay_cycles(500000);//
424  // Loop until XT1,XT2 & DCO fault flag is cleared
425  do
426  {
427  UCSCTL7 &= ~(XT2OFFG + XT1LFOFFG + DCOFFG); // Clear XT2,XT1,DCO fault flags
428  SFRIFG1 &= ~OFIFG; // Clear fault flags
429  }while (SFRIFG1&OFIFG); // Test oscillator fault flag

433  uint32_t PMMRIE_backup, SVSMHCTL_backup, SVSMLCTL_backup;
434 
435  //The code flow for increasing the Vcore has been altered to work around
436  //the erratum FLASH37.
437  //Please refer to the Errata sheet to know if a specific device is affected
438  //DO NOT ALTER THIS FUNCTION
439 
440  //Open PMM registers for write access
441  PMMCTL0_H = 0xA5;
442 
443  //Disable dedicated Interrupts
444  //Backup all registers
445  PMMRIE_backup = PMMRIE;
446  PMMRIE &= ~(SVMHVLRPE | SVSHPE | SVMLVLRPE |
447  SVSLPE | SVMHVLRIE | SVMHIE |
448  SVSMHDLYIE | SVMLVLRIE | SVMLIE |

452  SVSMLCTL_backup = SVSMLCTL;
453 
454  //Clear flags
455  PMMIFG = 0;
456 
457  //Set SVM highside to new level and check if a VCore increase is possible
458  SVSMHCTL = SVMHE | SVSHE | (SVSMHRRL0 * level);
459 
460  //Wait until SVM highside is settled
461  while((PMMIFG & SVSMHDLYIFG) == 0)
462  {
463  ;
464  }
465 
466  //Clear flag

469  //Check if a VCore increase is possible

472  //-> Vcc is too low for a Vcore increase
473  //recover the previous settings
474  PMMIFG &= ~SVSMHDLYIFG;
475  SVSMHCTL = SVSMHCTL_backup;
476 
477  //Wait until SVM highside is settled
478  while((PMMIFG & SVSMHDLYIFG) == 0)
479  {
480  ;
481  }
482 
483  //Clear all Flags
484  PMMIFG &= ~(SVMHVLRIFG | SVMHIFG | SVSMHDLYIFG |
485  SVMLVLRIFG | SVMLIFG |
486  SVSMLDLYIFG
487  );
488

490  PMMRIE = PMMRIE_backup;
491  //Lock PMM registers for write access
492  PMMCTL0_H = 0x00;
493  //return: voltage not set
494  return false;

498  //Vcc is high enough for a Vcore increase

521  }

526 
527  //Restore Low side settings
528  //Clear all other bits _except_ level settings
529  SVSMLCTL &= (SVSLRVL0 + SVSLRVL1 + SVSMLRRL0 +

531  );

536 
537  //Restore low-side SVS monitor settings
538  SVSMLCTL |= SVSMLCTL_backup;
539 
540  //Restore High side settings
541  //Clear all other bits except level settings

543  SVSMHRRL0 + SVSMHRRL1 +
544  SVSMHRRL2
545  );
546
